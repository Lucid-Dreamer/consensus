\TITLE{The GM}

\SECTION{Agenda}
\vspace{-1.5em}
\textbf{
 \begin{itemize}
  \item Make the players' characters' lives full of challenges,
    obstacles, and some opportunity
  \item Play to find out what happens
  \item Ask what it means to be human
  \item Make the Consensus real
  \end{itemize}
}

\SECTION{Always Say}
\begin{itemize}
\item What the principles demand.
\item What the rules demand
\item What your prep demands
\item What honesty demands
\end{itemize}

\SECTION{The Principles}
\begin{itemize}
\item Riddle the world with hidden magic, unexpected connections, and
  unrecognized potential
\item Address yourself to the characters, not the players
\item Make your move, but misdirect
\item Make your move, but never speak its name
\item Look down the sights
\item Name everyone, make everyone human
\item Ask provocative questions and build on the answers
\item Sometimes, give them exactly what theyâ€™re asking for
\item Be a fan of the players' characters
\item Think offscreen too
\item Sometimes, declaim decision-making
\end{itemize}

\SECTION{Your Moves}
\begin{itemize}
\item Separate them
\item Capture someone
\item Put someone in a spot
\item Deal damage
\item Advance a Status Track
\item Hit them with the static realities of their world
\item Announce off-screen badness
\item Show them the downsides of their paradigm
\item Announce future badness
\item Assign Discord
\item Make them Backlash
\item Take away their stuff
\item Threaten an Ancho
\item Change the Battlefield
\item Make them search
\item Introduce a Lookout
\item Tell them the possible consequences and ask
\item Offer an opportunity, with or without a cost
\item Turn their move back on them
\item Make a front move (from one of your fronts)
\item After every move: ``what do you do?''
\end{itemize}
